49class ItemsListItem {
50public:
51 enum type {
52 item_is_geometry,
53 item_is_list
54 };
55
56 ItemsListItem(void* item_)
57 : t(item_is_geometry)
58 {
59 item.g = item_;
60 }
61 ItemsListItem(ItemsList* item_)
62 : t(item_is_list)
63 {
64 item.l = item_;
65 }
66
67 type
68 get_type() const
69 {
70 return t;
71 }
72
73 void*
74 get_geometry() const
75 {
76 assert(t == item_is_geometry);
77 return item.g;
78 }
79 ItemsList*
80 get_itemslist() const
81 {
82 assert(t == item_is_list);
83 return item.l;
84 }
85
86 type t;
87 union {
88 void* g;
89 ItemsList* l;
90 } item;
91};

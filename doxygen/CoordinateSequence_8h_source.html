440  template<typename T>
441  void add(const T& c, std::size_t pos)
442  {
443  static_assert(std::is_base_of<CoordinateXY, T>::value, "Must be a Coordinate class");
444 
445  // c may be a reference inside m_vect, so we make sure it will not
446  // grow before adding it
447  if (m_vect.size() + stride() <= m_vect.capacity()) {
448  make_space(pos, 1);
449  setAt(c, static_cast<std::size_t>(pos));
450  } else {
451  T tmp{c};
452  make_space(pos, 1);
453  setAt(tmp, static_cast<std::size_t>(pos));
454  }

466  template<typename T>
467  void add(std::size_t i, const T& coord, bool allowRepeated)
468  {
469  // don't add duplicate coordinates
470  if(! allowRepeated) {
471  std::size_t sz = size();
472  if(sz > 0) {
473  if(i > 0) {
474  const CoordinateXY& prev = getAt<CoordinateXY>(i - 1);
475  if(prev.equals2D(coord)) {
476  return;
477  }
478  }
479  if(i < sz) {
480  const CoordinateXY& next = getAt<CoordinateXY>(i);
481  if(next.equals2D(coord)) {
482  return;
483  }
484  }
485  }
486  }
487 
488  add(coord, i);
489  }

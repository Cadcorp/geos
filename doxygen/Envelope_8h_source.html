675  double distanceSquared(const Envelope& env) const
676  {
677  double dx = std::max(0.0,
678  std::max(maxx, env.maxx) - std::min(minx, env.minx) - (maxx - minx) -
679  (env.maxx - env.minx));
680  double dy = std::max(0.0,
681  std::max(maxy, env.maxy) - std::min(miny, env.miny) - (maxy - miny) -
682  (env.maxy - env.miny));
683 
684  return dx * dx + dy * dy;
685  };

713  static double distanceSquaredToCoordinate(
714  const CoordinateXY& c,
715  const CoordinateXY& p0,
716  const CoordinateXY& p1)
717  {
718  double xa = c.x - p0.x;
719  double xb = c.x - p1.x;
720  double ya = c.y - p0.y;
721  double yb = c.y - p1.y;
722 
723  // If sign of a and b are not the same, then Envelope spans c and distance is zero.
724  double dx = (std::signbit(xa) == std::signbit(xb)) * std::min(std::abs(xa), std::abs(xb));
725  double dy = (std::signbit(ya) == std::signbit(yb)) * std::min(std::abs(ya), std::abs(yb));
726 
727  return dx*dx + dy*dy;
728  }

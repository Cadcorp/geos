328  template <class T>
329  std::unique_ptr<Geometry>
330  buildGeometry(T from, T toofar) const
331  {
332  bool isHeterogeneous = false;
333  std::size_t count = 0;
334  int geomClass = -1;
335  for(T i = from; i != toofar; ++i) {
336  ++count;
337  const Geometry* g = *i;
338  if(geomClass < 0) {
339  geomClass = g->getSortIndex();
340  }
341  else if(geomClass != g->getSortIndex()) {
342  isHeterogeneous = true;
343  }
344  }
345 
346  // for the empty geometry, return an empty GeometryCollection
347  if(count == 0) {
348  return std::unique_ptr<Geometry>(createGeometryCollection());
349  }
350 
351  // for the single geometry, return a clone
352  if(count == 1) {
353  return (*from)->clone();
354  }
355 
356  // Now we know it is a collection
357 
358  // FIXME:
359  // Until we tweak all the createMulti* interfaces
360  // to support taking iterators we'll have to build
361  // a custom vector here.
362  std::vector<std::unique_ptr<Geometry>> fromGeoms;
363  for(T i = from; i != toofar; ++i) {
364  fromGeoms.push_back((*i)->clone());
365  }
366 
367  // for an heterogeneous ...
368  if(isHeterogeneous) {
369  return createGeometryCollection(std::move(fromGeoms));
370  }
371 
372  // At this point we know the collection is not hetereogenous.
373  switch((*from)->getDimension()) {
374  case Dimension::A: return createMultiPolygon(std::move(fromGeoms));
375  case Dimension::L: return createMultiLineString(std::move(fromGeoms));
376  case Dimension::P: return createMultiPoint(std::move(fromGeoms));
377  default:
378  throw geos::util::IllegalArgumentException(std::string("Invalid geometry type."));
379  }
380  }

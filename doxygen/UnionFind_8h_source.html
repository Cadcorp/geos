62 size_t find(size_t i) {
63 size_t root = i;
64
65 while(clusters[root] != root) {
66 root = clusters[root];
67 }
68
69 while (i != root) {
70 size_t next = clusters[i];
71 clusters[i] = root;
72 i = next;
73 }
74
75 return i;
76 }

83 void join(size_t i, size_t j) {
84 auto a = find(i);
85 auto b = find(j);
86
87 if (a == b) {
88 return;
89 }
90
91 if ((sizes[b] > sizes[a]) || (sizes[a] == sizes[b] && b <= a)) {
92 std::swap(a, b);
93 }
94
95 clusters[a] = clusters[b];
96 sizes[b] += sizes[a];
97 sizes[a] = 0;
98
99 num_clusters--;
100 }
